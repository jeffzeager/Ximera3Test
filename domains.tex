\begin{problem}
Find the implied domain of $f(x) = \frac{x - 1}{x + 3}$
\begin{multipleChoice}
\choice{$(-\infty, 1) \cup (1, \infty)$}
\choice{$\{-3, 1\}$}
\choice[correct]{$(-\infty, -3) \cup (-3, \infty)$}
\choice{$(-\infty, -3) \cup (-3, 1) \cup (1, \infty)$}
\choice{None of the above}
\end{multipleChoice}
\end{problem}
